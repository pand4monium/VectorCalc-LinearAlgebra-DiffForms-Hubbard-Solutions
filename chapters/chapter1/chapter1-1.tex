\bx{ % 1.1.2
    Compute the following vectors:
    \eah{
        \item $\bmat{3 \\ \pi \\ 1} + \bmat{1 \\ -1 \\ \sqrt2}$
        \item $\bmat{1 \\ 4 \\ c \\ 2} + \bvec e_2$
        \item $\bmat{1 \\ 4 \\ c \\ 2} - \bvec e_4$
    }
    \sol{
        This is simple. \\
        \eah*{
            \item $\bmat{3 \\ pi \\ 1} + \bmat{1 \\ -1 \\ \sqrt2} = \bmat{4 \\ \pi -1 \\ 1 + \sqrt 2}$
            \item $\bmat{1 \\ 4 \\ c \\ 2} + \bvec{e_2} = \bmat{1 \\ 5 \\ c \\ 2}$
            \item $\bmat{1 \\ 4 \\ c \\ 2} -\bvec{e_4} = \bmat{1 \\ 4 \\ c \\ 1}$
        }
    }
}

\bx{ % 1.1.8
    Suppose that water is flowing through a pipe of radius $r$, with speed $r^2 -a^2$, where $a$ is the distance to the axis of the pipe.
    \ea{
        \item Write a vector field describing the flow if the pipe is the direction of the $z$-axis.
        \item Write the vector field describing the flow if the axis of the pipe is the unit circle in the $(x, y)$-plane
    }

    \sol{
        Due to constraints, the vector fields will not be drawn. \\
        \eah*{
            \item $\defvF{x \\ y \\ z}{0 \\ 0 \\ r^2 - x^2 - y^2}$
            \item $\vF \pmat{x \\ y \\ z} = \left( r^2 - \left( \left(\sqrt{x^2 + y^2} - 1\right)^2 + z^2 \right)\right)  \bmat{-\frac{y}{\sqrt{x^2+y^2}} \\ \frac{x}{\sqrt{x^2+y^2}} \\ 0}$
            }
        }
    }


\bx{ % 1.1.9
    Show that the set of complex numbers $\{z|\mathrm{Re}(wz) = 0\}$ with fixed $w\in\C$ is a subspace of $\R^2 = \C$. Describe this subspace.
    \sol{
        Let $S = \{z|\mathrm{Re} (wz) = 0\}$. We will first show that it is closed under addition and scalar multiplications. Firstly, $0 \in S$ since $\mathrm{Re}(w \cdot 0) = \mathrm{Re}(0) = 0$. Hence the subspace is non-empty. 
        
        Next, let $c \in \R$ and $x,y\in S$. We will show that $cx + y \in S$. 
        \[
        \mathrm{Re}((cx+y)w) = c\mathrm{Re}(xw) + \mathrm{Re}(yw) = c(0) + 0 = 0
        \]
        And we are done.
    }
}
