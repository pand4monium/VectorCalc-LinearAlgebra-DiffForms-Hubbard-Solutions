\section{Introducing the actors: Matrices}
 
\bx{ % 1.2.1
    \ea{
        \item What are the dimensions of the following matrices? \eah{
            \item $\bmat{a & b & c \\ d & e & f}$
            \item $\bmat{4 & 1 \\ 0 & 2}$
            \item $\bmat{\pi & 1 \\ 0 & 1 \\ 1 & 0}$
            \item $\bmat{1 & 0 & 0 & 1 \\ 0 & 1 & 0 & 1 \\ 1 & 0 & 1 & 0}$
            \item $\bmat{1 & 0 & 0 \\ 0 & 1 & 0 \\ 0 & 0 & 1}$
        }
        \item Which of the preceding matrices can be multiplied together?
    }
}

\bx{ % 1.2.2
    Perform the following matrix multiplications when possible.
    \eag{3}{
        \task\ $\bmat{1 & 2 & 3 \\ 4 & 5 & 6} \bmat{7 & 8 \\ 9 & 10 \\ 1 & 2}$
        \task\ $\bmat{1 & 2 \\ 0 & 3} \bmat{1 & 4 \\ -1 & 3 \\ -2 & 2}$
        \task\ $\bmat{1 & -1 & 1 \\ -1 & 0 & 2 \\ -1 & 1 & 1} \bmat{0 & 1 & -1 \\ -1 & 1 & 2 \\ 2 & 0 & -2}$
        \task\ $\bmat{7 & 1 \\ -1 & 0 \\ 2 & 3} \bmat{5 \\ -4}$
        \task\ $\bmat{1 & 2 \\ 0 & 3} \bmat{1 & 4 \\ -1 & 3} \bmat{0 & 1 \\ -1 & 3}$
        \task\ $\bmat{0 & 2 & 1 \\ 1 & 3 & 2}\bmat{0 & 1 \\ 3 & 5}$
    }
}

\bx{ % 1.2.3
    Given the matrices $A$ and $B$ in the margin at left,
    \[
    A = \bmat{1 & 2 & 0 \\ 3 & 1 & -1} \quad 
    B = \bmat{2 & 5 & 1 \\ 1 & 4 & 2 \\ 1 & 3 & 3}
    \]
    \ea{
        \item compute the third column of $AB$ \textit{without} computing the entire matrix $AB$.
        \item compute the second row of $AB$, again \textit{without} computing the entire matrix $AB$.
    }
}

\bx{ % 1.2.4
    Compute the following without doing any arithmetic.
    \eah{
        \item $\bmat{7 & 2 & \sqrt3 & 4 \\ 6 & 8 & a^2 & 2 \\ 3 & \sqrt5 & a & 7} \bmat{0 \\ 1 \\ 0 \\ 0}$
        \item $\bmat{6a & 2 & 3a^2 \\ 4 & 2\sqrt a & 2 \\ 5 & 12 & 3} \bvec e_2$
        \item $\bmat{2 & 1 & 8 & 6 \\ 3 & 2 & \sqrt 3 & 4} \bvec e_3$
    }
}

\bx{ % 1.2.5
    Let $A$ and $B$ be $n \times n$ matrices, with $A$ symmetric. Are the following true or false?
    \eah{
        \item $\tpse{(AB)} = \tpse BA$
        \item $\tpse{(\tpse A B)} = \tpse B \tpse A$
        \item $\tpse{(\tpse A B)} = BA$
        \item $\tpse{(AB)} = \tpse A \tpse B$
    }
}

\bx{ % 1.2.6
    Which of the following matrices are diagonal? symmetric? triangular? antisymmetric? Do the matrix multiplication by hand.
    \eag{4}{
        \task\ $\bmat{a & 0 \\ 0 & a}$
        \task\ $\bmat{a&0\\0&a}^2$
        \task\ $\bmat{a&0\\0&a}\bmat{0&0\\b&b}$
        \task\ $\bmat{a&0\\0&b}^2$
        \task\ $\bmat{0&0\\a&a}^2$
        \task\ $\bmat{0&0\\a&a}^3$
        \task\ $\bmat{1&0\\1&1}\bmat{1&0\\-1&1}$
        \task\ $\bmat{1&1&-1\\1&0&1\\-1&1&0}$
        \task\ $\bmat{1&0\\-1&0}^3$
        \task\ $\bmat{1&0&1\\0&1&0\\1&0&1}^2$
        \task\ $\bmat{1&0&-1\\0&1&0\\1&0&1}^2$
        \task\ $\bmat{1&0\\-1&0}^4$
    }
}
