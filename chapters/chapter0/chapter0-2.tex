\section{Quantifiers and Negation}

\bx{
    Negate the following statements: \ea{
        \item Every prime number such that if you divide it by 4 you have a remainder of 1 is the sum of two squares.
        \item For all $x \in \mathbb R$ and for all $\epsilon > 0$, there exists $\delta > 0$ such that for all $y \in \mathbb R$, if $|y -x| < \delta$, then $|y^2 - x^2| < \epsilon$.
        \item For all $\epsilon > 0$, there exists $\delta > 0$ such that for all $x, y \in \mathbb R$, if $|y - x| < \delta$, then $|y^2 - x^2| < \epsilon$.
    }
    \sol{\ea{
        \item There exists a prime number in the form of $4n + 1$, for $n \in \mathbb N_0$ that cannot be expressed as the sum of two squares.
        \item There exists $x \in \mathbb R$ and $\epsilon > 0$ such that for all $\delta > 0$, there exists $ y \in \mathbb R$ with $|y-x| < \delta$ but $|y^2 - x^2| \ge \epsilon$.
        \item There exists $\epsilon > 0$ such that for all $\delta > 0$, there exists $x, y \in \mathbb R$ with $|y-x| < \delta$ but $|y^2 - x^2| \ge \epsilon$.
    }}
}
