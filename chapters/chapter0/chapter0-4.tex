    Are the following true functions? That is, are they both everywhere defined and well defined? \ea{
        \item "The aunt of", from people to people.
        \item $f(x) = \frac 1x$, from real numbers to real numbers.
        \item "The capital of", from countries to cities (careful - at least one country, Bolivia, has to capitals.)
    }
    \sol{
        \ea{
            \item Considering the cases where and individual's parents not having any sisters, the individual does not have an aunt, making it not everywhere defined. Considering the cases where an individual has multiple aunts, the function is not well defined.
            \item The function is not everywhere defined on real numbers as when $x = 0$, $f(x)$ is undefined. The function however is wel defined as if $f(x) = f(y)$, then $x =y$.
            \item The function is everywhere defined as every country has a capital. However, it is not well defined as some countries, such as Bolivia, has two capitals.
        }
    }
}

\bx{ % 0.4.6
    Prove Proposition 0.4.11. \textbf{Inverse image of intersection, union} \ea{
        \item The inverse image of an intersection equals the intersection of the inverse images: \[
        f^{-1} (A \cap B) = f^{-1} (A) \cap f^{-1} (B)\]
        \item The inverse image of a union equals the union of the inverse images: \[
        f^{-1} (A \cup B) = f^{-1} (A) \cup f^{-1} (B)\]
    }

    \sol{\ea{
        \item We will show this by inclusion both ways: \er{
            \item Let $y \in A \cap B$ with $x = f^{-1}(y) \in f^{-1}(A \cap B)$. Then $y \in A$ and $y \in B$, implying $x \in f^{-1}(A)$ and $x \in f^{-1}(B)$, thus $x \in f^{-1}(A) \cap f^{-1}(B)$.
            \item The inverse is similar. $x \in f^{-1}(A) \cap f^{-1}(B)$, then $x \in f^{-1} (A)$ and $x \in f^{-1} (B)$, so $y \in A$ and $y \in B$. This shows that $y \in A \cap B$, hence, $x \in f^{-1}(A \cap B)$.
        }
        \item Similarly, by inclusion both ways: \er{
            \item Let $y \in A \cup B$ with $x = f^{-1}(y) \in f^{-1}(A \cup B)$. Then $y \in A$ or $y \in B$, implying $x \in f^{-1}(A)$ or $x \in f^{-1}(B)$, thus $x \in f^{-1}(A) \cup f^{-1}(B)$.
            \item The inverse is similar. $x \in f^{-1}(A) \cup f^{-1}(B)$, then $x \in f^{-1} (A)$ or $x \in f^{-1} (B)$, so $y \in A$ or $y \in B$. This shows that $y \in A \cup B$, hence, $x \in f^{-1}(A \cup B)$.
        }
    }}
}

\bx{ % 0.4.7
    Evaluate $(f \circ g \circ h) (x)$ at $x = a$ for the following. \ea{
        \item $f(x) = x^2 -1, \quad g(x) = 3x, \quad h(x) = -x+ 2, \quad \text{for } a = 3$. 
        \item $f(x) = x^2 , \quad g(x)=x-3, \quad h(x) = x-3, \quad \text{for } a= 1$
    }
    \sol{\ea{
        \item \[
        \begin{aligned}
        (f\circ g \circ h)(3) &= f(g(h(3))) \\
        &= f(g(-3 + 2)) = f(g(-1)) \\
        &= f(3(-1)) = f(-3) \\
        &= (-3)^2 - 1 = 8
        \end{aligned}
        \]
        \item \[
        \begin{aligned}
        (f\circ g \circ h)(1) &= f(g(h(1))) \\
        &= f(g(1-3)) = f(g(-2)) \\
        &= f(-2-3) = f(-5) \\
        &= (-5)^2 = 25
        \end{aligned}
        \]
    }}
}

\bx{ % 0.4.8
    \ea{
        \item Prove Proposition 0.4.16. \textbf{(Composition of onto functions).} Let the functions $f : B \rightarrow C$ and $g : A \rightarrow B$ be onto. Then the composition $(f \circ g)$ is onto.
        \item Prove Proposition 0.4.17. \textbf{(Composition of one to one functions).} Let $f: B\rightarrow C$ and $g: A \rightarrow B$ be one to one. Then the composition $(f\circ g)$ is one to one.
    }
    \sol{
    \ea{
        \item Let $c \in C$. Since $f$ is onto, there exists some $b \in B$ such that $f(b)=c$. Since $g$ is onto, there exists some $a \in A$ such that $g(a)=b$. Hence $(f\circ g)(a)=f(g(a))=f(b)=c$. Since every $c\in C$ has a preimage under $f\circ g$, the composition $f\circ g$ is onto.

        \item Let $a_1,a_2\in A$ and suppose $(f\circ g)(a_1)=(f\circ g)(a_2)$. Then $f(g(a_1))=f(g(a_2))$. Since $f$ is one to one, $g(a_1)=g(a_2)$. Since $g$ is one to one, $a_1=a_2$. Therefore $f\circ g$ is one to one.
    }
}
}

\bx{ % 0.4.9
    What is the natural domain of $\displaystyle \sqrt{\frac xy}$?
    \sol{
        For $\sqrt{\frac xy}$ to be defined, $\frac xy \ge 0$. Hence the natural domain is $\{(x, y) \in \mathbb R^2 : y \ne 0, \frac xy \ge 0\}$.
    }
}

\bx{ % 0.4.10
    What is the natural domain of \eah{
        \item $\ln \circ \ln $? 
        \item $\ln \circ \ln \circ \ln $? 
        \item $\ln$ composed with itself $n$ times?
    }
    \sol{ \ea{
        \item The domain of $\ln x$ is $x > 0$. Hence the domain of $\ln \circ \ln$ is when $\ln x > 0$ which is $\{x \in \mathbb R: x > 1 \}$.
        \item Similarly, the natural domain of $\ln\circ\ln\circ\ln$ is $\{x\in \mathbb R: x >e\}$.
        \item The natural domain of $\ln$ composed with itself $n$ times is $\{x \in \mathbb R :x > e \uparrow \uparrow (n-2) \}$.
    }}
}

\bx{ % 0.4.11
    What subset of $\mathbb R$ is the natural domain of the function $(1+x)^{1/x}$?
    \sol{
        For $(1+x)^{1/x}$ to be defined, we must have the following restrictions, $(1+x) >0$; to avoid fractional powers of negative numbers, and $x \ne 0$; as we cannot divide by zero. Hence, the natural domain is $(1, \infty) \setminus \{0\}$.
    }
}

\bx{ % 0.4.12
    The function $f(x) = x^2$ from real numbers to real nonnegative numbers is onto but not one to one.\ea{
        \item Can you make it one to one by changing its domain? By changing its codomain?
        \item Can you make it not onto by changing its domain? Its codomain?
    }
    \sol{\ea{
        \item Yes.
        \er{
            \item Change the domain to $[0,\infty)$. Then $f(x)=x^2$ is one to one and remains onto $\mathbb{R}_{\ge 0}$.
            \item Changing the codomain to $\{0\}$, makes $f(x) = x^2$ trivially one to one
        }

        \item Yes.
        \er{
            \item Change the domain to $(0,\infty)$. Then $0$ is no longer attained, so $f$ is not onto $\mathbb{R}_{\ge 0}$.
            \item Change the codomain to $\mathbb{R}$. Then negative real numbers are not attained, so $f$ is not onto.
        }
    }}
