\section{Real Numbers}

\bx{ % 0.5.1
    Show that if $p$ is a polynomial of odd degree with real coefficients, then thereis a real numbr $c$ such that $p(c) = 0$
}

\bx{ % 0.5.2
    \ea{
        \item Show that the function \[
            f(x) = \begin{cases} 
                \sin \frac1x & \text{if } x \ne 0 \\
                0 & \text{if } x = 0 
            \end{cases} 
            \text{ is not continuous.}
            \]
        \item Show that $f$ satisfies the conclusion of the intermediate value theorem: If $f(x_1)=a_1$ and $f(x_2) = a_2$, thn for any number $a$ between $a_1$ and $a_2$, there exists a number $x$ between $x_1$ and $x_2$ such that $f(x) = a$.
    }
}

\bx{ % 0.5.3
    Suppose $a \le b$. Show that if $f \ : \ [a,b] \rightarrow [a,b]$ is continuous, there exists $c \in [a, b]$ with $f(c) = c$.
}

\bx{ % 0.5.4
    Let \[
    a_n = \frac{(-1)^{n+1}}n, \quad \text{for } n = 1, 2, \dots
    \] \ea{
        \item Show that the series $\sum a_n$ is convergent.
        \item Show that $\sum_{n=1}^\infty = \ln 2$.
        \item Explain how to rearrange the terms of the series so it converges to 5.
        \item Explain how to rearrange the terms of the series so that it diverges.
    }
}

\bx{ % 0.5.5
    Show that if a series $\sum_{k=1}^\infty a_n$ is absolutely convergent, then any rearrangement of the series is still converget and converges to the same limit. \textit{Hint: } For any $\epsilon > 0$, there exists $N$ such that $\sum_{n=N+1}^\infty < \epsilon$. For any rearrangement $\sum_{k=1}^\infty b_n$ of the series, there exists $M$ such that all of $a_1, \dots, a_N$ appear among $b_1, \dots, b_M$. Show that $\left|\sum_{n=1}^\infty a_n - \sum_{n=1}^M b_n \right| < \epsilon$.
}
